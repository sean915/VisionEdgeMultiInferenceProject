    int Detector::pushFrame(const FrameData& frame) {
        if (status_ != AlgorithmStatus::READY && status_ != AlgorithmStatus::RUNNING)
            return -1;

        // ⚠️ 가정: 입력은 BGR 8UC3, 연속 메모리
        // 실제 포맷이 다르면 여기에서 cvType/stride/bytesPerPixel을 맞춰야 함.
        const int cvType = CV_8UC3;
        const int bytesPerPixel = 3;
        const int stride = frame.width * bytesPerPixel;
        const size_t totalBytes = static_cast<size_t>(stride) * static_cast<size_t>(frame.height);

        OwnedFrame of;
        of.index = frame.index;
        of.width = frame.width;
        of.height = frame.height;
        of.cvType = cvType;
        of.strideBytes = stride;
        of.timestamp = frame.timestamp;
        of.buf.resize(totalBytes);

        // frame.data는 호출자 메모리. 여기서 즉시 복사해서 DLL이 소유하도록!
        if (!frame.data || frame.width <= 0 || frame.height <= 0) return -1;
        std::memcpy(of.buf.data(), frame.data, totalBytes);

        frame_queue_owned_.push(std::move(of));
        return 0;
    }

    static cv::Mat make_crop_cpp(const cv::Mat& frame_bgr,
        HMSTACK::Detector::CropMode mode,
        const cv::Rect& cell,
        const cv::Rect& pnp,
        int padding)
    {
        int w = frame_bgr.cols, h = frame_bgr.rows;
        auto clampRect = [&](cv::Rect r) {
            int x1 = std::max(0, r.x - padding);
            int y1 = std::max(0, r.y - padding);
            int x2 = std::min(w, r.x + r.width + padding);
            int y2 = std::min(h, r.y + r.height + padding);
            if (x2 <= x1) x2 = std::min(w, x1 + 1);
            if (y2 <= y1) y2 = std::min(h, y1 + 1);
            return cv::Rect(x1, y1, x2 - x1, y2 - y1);
            };

        if (mode == HMSTACK::Detector::CropMode::FULL) return frame_bgr.clone();
        if (mode == HMSTACK::Detector::CropMode::PNP)  return frame_bgr(clampRect(pnp)).clone();
        return frame_bgr(clampRect(cell)).clone(); // default CELL
    }

    static DefectEvalResult evaluate_defect_cpp(float p_ab, float p_no, float p_em, float sure, float min_valid)
    {
        DefectEvalResult result;
        // 가장 높은 확률과 클래스 결정
        float probs[3] = { p_ab, p_no, p_em };
        const char* names[3] = { "Abnormal", "Normal", "Empty" };
        int top1_idx = 0;
        for (int i = 1; i < 3; ++i) {
            if (probs[i] > probs[top1_idx]) top1_idx = i;
        }
        result.top1 = names[top1_idx];
        result.top1_score = probs[top1_idx];

        // 정책: 확신(sure) 이상이면 확정, min_valid 이상이면 Questionable, 아니면 Error
        if (result.top1_score >= sure) {
            result.final = result.top1;
            result.reason = "High confidence";
        }
        else if (result.top1_score >= min_valid) {
            result.final = "Questionable";
            result.reason = "Low confidence";
        }
        else {
            result.final = "Error";
            result.reason = "Below minimum valid threshold";
        }
        return result;
    }
